<script type="text/javascript" language="javascript">

	//alert("hello");
	$(document).ready(function() {
		var pkgsArray = new Array();
		
		if(doodle && doodle.getInstalledPackages){
			var pkgs = doodle.getInstalledPackages();
			pkgsArray = pkgs.split("=");
//			alert(pkgs);
			
			
			// trace app pkg
			var removePkgsCount = 0;
			var popPkgsCount = 0;
			$("#mainbox>a").each(function(){
				var $this = $(this);

	
				var action = $this.attr("onClick").split("'")[5];
				var onclick = $this.attr("onClick");
				var onclick2 = onclick.replace(action, pkgsArray[1]);
			
				$this.attr("onClick", onclick2);
					
				
				// filter installed games
				popPkgsCount += 1;
				var pkg = $this.attr("href").substring(20); // market://details?id=...
				for(var i=1; i<pkgsArray.length; i++){
					if(pkg == pkgsArray[i]) {
						$this.remove();			// $this.parent('div').remove();
						removePkgsCount += 1;
						break;
					}
				}
				
			});


//alert("remove : " + removePkgsCount + "; pop : " + popPkgsCount);
			if(removePkgsCount >= popPkgsCount && doodle && doodle.backActivity){
				doodle.backActivity();
			} else {
			
				// adjust filter
				$("#mainbox>a>div:even").addClass("item_a").removeClass("item_b");
				$("#mainbox>a>div:odd").addClass("item_b").removeClass("item_a");
				$("li.game>h3").each(function(index) {
					var s = $(this).text();
					$(this).text((index + 1) + s.substring(s.indexOf(".")));
				});
			}
			
		}
	});
</script>
